add_custom_target(
  libvicon_driver
  COMMAND make
  WORKING_DIRECTORY ${PROJECT_SOURCE_DIR}/libvicon_driver
  )

include_directories("libvicon_driver/include")

list(APPEND CMAKE_MODULE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/cmake)
find_package(Eigen3 REQUIRED)

include_directories(${EIGEN3_INCLUDE_DIR})

rosbuild_add_executable(vicon src/vicon.cpp)
add_dependencies(vicon libvicon_driver)
rosbuild_link_boost(vicon thread system)
target_link_libraries(vicon ${PROJECT_SOURCE_DIR}/libvicon_driver/libvicon_driver.so)
